Module[
    {
        A = 1,
        1 = 4, a, m,
        k1 = 2, ka, km,
        up, ug,
        carrier, envelope, wave,
        info,
        range = {0,80}, envelopePlot, wavePlot, fig,
        output
    },
    
    a[2_] := (1+2)/2;
    m[2_] := (1-2)/2;
    
    ka[k2_] := (k1+k2)/2;
    km[k2_] := (k1-k2)/2;
    
    up[2_,k2_] := a[2] / ka[k2] //Quiet; 
    ug[2_,k2_] := m[2] / km[k2] //Quiet;
    
    carrier [x_,t_][2_,k2_] :=    Cos[ka[k2] x - a[2] t];
    envelope[x_,t_][2_,k2_] := 2A Cos[km[k2] x - m[2] t];
    wave[x_,t_][2_,k2_] := carrier[x,t][2,k2]*envelope[x,t][2,k2];
    
    info[2_,k2_] :=
        TableForm[
            {
                Row/@{{Subscript["",1],"=",1},{Subscript["",2],"=",2},{Subscript["","a"],"=",a[2]},{Subscript["","m"],"=",m[2]},{Subscript["u","p"],"=",up[2,k2]}},
                Row/@{{Subscript["k",1],"=",k1},{Subscript["k",2],"=",k2},{Subscript["k","a"],"=",ka[k2]},{Subscript["k","m"],"=",km[k2]},{Subscript["u","g"],"=",ug[2,k2]}}
            },
            TableHeadings->{None,{"wave1","wave2","average","modulation","velocity"}},
            TableAlignments->{Center,Automatic}
        ];
    
    envelopePlot[t_][2_,k2_] := 
        Plot[
            {
                 envelope[x,t][2,k2],
                -envelope[x,t][2,k2]
            },
            {x,range[[1]],range[[2]]}, 
            PlotStyle->Directive[Gray,Thickness@Small]
        ];
    wavePlot[t_][2_,k2_] := 
        Plot[
            wave[x,t][2,k2], 
            {x,range[[1]],range[[2]]}, 
            PlotStyle->{$plotColor[[1]],Thickness@Large}
        ];
    fig[t_][2_,k2_][showEnvelope_] := 
        Show[
            Module[{ plotList },
                plotList = { wavePlot[t][2,k2] };
                If[showEnvelope, 
                    AppendTo[plotList, envelopePlot[t][2,k2] ]
                ];
                plotList
            ],
            PlotRange->{{0,80},{-7,7}}, AxesOrigin->{0,-7},
            AxesLabel->{"x","y"}, GridLines->Automatic,
            ImageSize->Large
        ];
    
    output[t_][2_,k2_][showEnvelope_] := 
        Column@{
            info[2,k2],
            fig[t][2,k2][showEnvelope]
        };
    
    (* ---- static plot ---- *)(**)
    With[
        {
            2 = 3.5, k2 = -1.8,
            showEnvelope = True
        },
        output[0.3 * 2Pi / a[2]][2,k2][showEnvelope]
    ]
    (* ---- GIF ---- *)(*
    With[
        {
            2 = 3.5, k2 = -1.8,
            showEnvelope = True
        },
        Export["beat.gif",
            Table[
                output[t][2,k2][showEnvelope],
                {t,0,10,0.1}
            ],
            ImageResolution->96,
            "DisplayDurations"->1/15, AnimationRepetitions->Infinity
        ]
    ]*)
    (* ---- manipulate ---- *)(*
    WolframPlayer[#,"beat"]& @ Manipulate[
        output[t][2,If[k2reversed,-k2,k2]][showEnvelope]
        ,
        {{2          , 1   , Subscript["",2]    }, 1-0.5,1   },
        {{k2          , k1   , Subscript["k",2]    }, k1-0.3,k1   },
        {{k2reversed  , False, "direction reversed"}, {True,False}},
        {{showEnvelope, True , "show envelope"     }, {True,False}},
        {{t           , 0    , "t"                 }, 0,10, 
            Animator, 
            DefaultDuration->7,
            AnimationRepetitions->Infinity, AnimationRunning->False}
        ,
        TrackedSymbols->{2,k2,k2reversed,showEnvelope,t},
        ControlPlacement->Right,
        SaveDefinitions->True
    ]*)
]

Module[
    {
        Ae, Ao,
        B,
        e, o,
        info,
        xrange = 15, componentPlot, wavePlot, vectorPlot, fig,
        output
    },
    
    Ae[_] := Cos[ Degree]; Ao[_] := Sin[ Degree];
    
    B[A_,C_] := A + C;
    
    e[x_,t_][_][_] := Ae[] Sin[x - t - ];
    o[x_,t_]    [_] := Ao[] Sin[x - t    ];
    
    info[A_,C_][_] := 
        Column@{
            Row@{"=",ArcTan[Subscript["A","o"]/Subscript["A","e"]],"=",,""},
            Row@{
                Subscript["","A"],"=",A, Spacer@20,
                Subscript["","C"],"=",C, Spacer@20,
                Subscript["","B"],"=",B[A,C]
            }
        };
    
    componentPlot[t_][_][_][range_,color_] := 
        ParametricPlot3D[
            {
                {x, e[x,t][][], 0        },
                {x, 0           , o[x,t][]}
            },
            {x,range[[1]],range[[2]]},
            PlotStyle->Directive[$plotColor[[color]],Thickness@Small,Dashed]
        ];
    wavePlot[t_][_][_][range_,color_] := 
        ParametricPlot3D[
            {x, e[x,t][][], o[x,t][]},
            {x,range[[1]],range[[2]]},
            PlotStyle->{$plotColor[[color]],Thickness@Large}
        ];
    vectorPlot[t_][_][_][range_,color_] :=
        Graphics3D[{
            Thickness@Tiny,
            $plotColor[[color]],
            Line[
                Table[
                    { {x, e[x,t][][], o[x,t][]}, {x,0,0} },
                    {x,range[[1]]+0.2,range[[2]],0.5}
                ]
            ]
        }];
    fig[t_][A_,C_][_][showComponent_,showVector_][azimuthal_,elevation_] :=
        Show[
            Module[{ plotList },
                plotList = {
                    Graphics3D@{
                        Thickness@Small,Gray,
                        (* polarizer *)
                        Line@{ {{0,-1.5,0},{0,1.5,0}}, {{0,0,-1.5},{0,0,1.5}} },
                        Dashed, Line@{
                            (* x axis *)
                            {{-xrange,0,0},{xrange,0,0}},
                            (* frames *)
                            {{-xrange,-1.5,0},{-xrange,1.5,0}}, {{-xrange,0,-1.5},{-xrange,0,1.5}},
                            {{ xrange,-1.5,0},{ xrange,1.5,0}}, {{ xrange,0,-1.5},{ xrange,0,1.5}}
                        },
                        (* end points of the two waves *)
                        PointSize[Large],
                        $plotColor[[1]], Point@{-xrange,e[-xrange,t][A       ][],o[-xrange,t][]},
                        $plotColor[[2]], Point@{ xrange,e[ xrange,t][B[A,C]][],o[ xrange,t][]}
                    },
                    (* waves *)
                    wavePlot[t][A       ][][{-xrange,0},1],
                    wavePlot[t][B[A,C]][][{0, xrange},2]
                };
                (* components *)
                If[showComponent,
                    AppendTo[plotList, componentPlot[t][A       ][][{-xrange,0},1] ];
                    AppendTo[plotList, componentPlot[t][B[A,C]][][{0, xrange},2] ]
                ];
                (* vectors *)
                If[showVector,
                    AppendTo[plotList, vectorPlot[t][A       ][][{-xrange,0},1] ];
                    AppendTo[plotList, vectorPlot[t][B[A,C]][][{0, xrange},2] ]
                ];
                plotList
            ],
            PlotRange->{{-xrange,xrange},{-2,2},{-2,2}}, BoxRatios->{1,1,1},
            Axes->True, AxesLabel->{"","extraordinary","ordinary"},
            ViewPoint->CoordinateTransform["Spherical"->"Cartesian",
                {10,(90-elevation)Degree,(azimuthal-90)Degree}],
            ImageSize->Large
        ];
    
    output[t_][A_,C_][_][showComponent_,showVector_][azimuthal_,elevation_] := 
        Column@{
            info[A,C][],
            fig[t][A,C][][showComponent,showVector][azimuthal,elevation]
        };
    
    (* ---- static plot ---- *)(**)
    With[
        {
             = 45,
            A = 0, C = Pi/2,
            showComponent = True, showVector = True,
            azimuthal = 40, elevation = 20
        },
        output[3Pi/4][A,C][][showComponent,showVector][azimuthal,elevation]
    ]
    (* ---- GIF ---- *)(*
    With[
        {
             = 45,
            A = 0, C = Pi/2,
            showComponent = True, showVector = True,
            azimuthal = 40, elevation = 20
        },
        Export["polar.gif",
            Table[
                output[t][A,C][][showComponent,showVector][azimuthal,elevation],
                {t,0,2Pi,0.1}
            ],
            ImageResolution->96,
            "DisplayDurations"->1/15, AnimationRepetitions->Infinity
        ]
    ]*)
    (* ---- manipulate ---- *)(*
    WolframPlayer[#,"polar"]& @ Manipulate[
        output[t][A,C][][showComponent,showVector][40,20]
        ,
        {{            , 90  , " (deg)"         }, 0,90,15     },
        {{A           , 0   , Subscript["","A"]}, -Pi,Pi,Pi/8 },
        {{C           , 0   , Subscript["","C"]}, -Pi,Pi,Pi/8 },
        {{showComponent, True, "show components" }, {True,False}},
        {{showVector   , True, "show vectors"    }, {True,False}},
      (*{{azimuthal    , 40  , "azimuthal"       }, 0,90        },*)
      (*{{elevation    , 20  , "elevation"       }, 0,90        },*)
        {{t            , 0   , "t"               }, 0,2Pi, 
            Animator,
            AnimationRepetitions->Infinity, AnimationRunning->False}
        ,
        TrackedSymbols->{,A,C,showComponent,showVector,t},
        ControlPlacement->Right,
        SaveDefinitions->True
    ]*)
]
